\section{Architettura Backend}
\begin{frame}{Il Ruolo del Server}
    Il backend opera in un ambiente \textbf{Trusted} (fido) con privilegi elevati.
    \begin{itemize}
        \item \textbf{Runtime}: Accesso diretto al File System, Variabili d'Ambiente e Rete.
        \item \textbf{Sicurezza}: Validazione centralizzata delle regole di business.
        \item \textbf{Agnosticismo}: Esposizione di dati (JSON) per client eterogenei.
    \end{itemize}
    \vspace{0.2cm}
    \textit{Strategia didattica: Persistenza su file JSON per focalizzarsi sui protocolli senza la complessità dei DBMS.}
\end{frame}

\begin{frame}{Runtime e Gestione Dipendenze}
    Lo sviluppo backend si basa su ecosistemi di pacchetti gestiti.
    \begin{itemize}
        \item \textbf{Python}: Gestione tramite \texttt{pip} e \textit{Virtual Environments}.
        \item \textbf{Node.js}: Gestione tramite \texttt{npm} e file \texttt{package.json}.
    \end{itemize}
    \vspace{0.3cm}
    \textbf{Hot Reloading}:
    \begin{itemize}
        \item Consente il riavvio automatico del server ad ogni modifica.
        \item Flask: \texttt{--debug} | Express: \texttt{nodemon}.
    \end{itemize}
\end{frame}

    L'handler associa una rotta a una funzione di gestione.
    \begin{itemize}
        \item \textbf{Pipeline}: Validazione $\rightarrow$ Esecuzione $\rightarrow$ Risposta.

    \end{itemize}

\section{REST API}
\begin{frame}{Modellazione delle Risorse}
    L'API REST si basa sulla manipolazione di entità logiche.
    \begin{itemize}
        \item \textbf{Risorse}: Sostantivi al plurale (es. \texttt{/tasks}, \texttt{/users}).
        \item \textbf{Verbi}: Metodi HTTP (es. \texttt{POST} per creare, \texttt{GET} per leggere).
        \item \textbf{Contratto}: Definizione chiara di input e output (JSON).
    \end{itemize}
\end{frame}

\begin{frame}{Todo App: Definizione Rotte}
    Esempio pratico: Gestione di una lista di attività (Task).
    \begin{itemize}
        \item \textbf{Recupero Lista}: \texttt{GET /tasks} $\rightarrow$ \texttt{200 OK}
        \item \textbf{Creazione}: \texttt{POST /tasks} $\rightarrow$ \texttt{201 Created}
        \item \textbf{Eliminazione}: \texttt{DELETE /tasks/<id>} $\rightarrow$ \texttt{204 No Content}
    \end{itemize}
    \vspace{0.3cm}
    \textit{Il backend deve supportare il parsing del corpo della richiesta (JSON).}
\end{frame}

\begin{frame}{Semantica REST e Status Codes}
    Corrispondenza tra operazioni ed esiti HTTP:
    \vspace{0.2cm}
    \centering
    \begin{tabular}{lll}
        \toprule
        \textbf{Metodo} & \textbf{Esito OK} & \textbf{Significato} \\
        \midrule
        \texttt{GET} & 200 & Successo \\
        \texttt{POST} & 201 & Risorsa Creata \\
        \texttt{PUT} & 200/204 & Aggiornamento \\
        \texttt{DELETE} & 204 & Cancellata \\
        - & 400 & Bad Request (Validazione) \\
        - & 404 & Not Found (Mancante) \\
        \bottomrule
    \end{tabular}
\end{frame}

\begin{frame}[fragile]{Implementazione REST (Flask)}
    Gestione della rotta POST con validazione minima.
    \begin{verbatim}
@app.route('/tasks', methods=['POST'])
def add_task():
    new_task = request.get_json()
    if not new_task or 'title' not in new_task:
        return jsonify({"error": "Bad Request"}), 400
    tasks.append(new_task)
    return jsonify({"status": "created"}), 201
    \end{verbatim}
\end{frame}

\begin{frame}[fragile]{Implementazione REST (Express)}
    Gestione della rotta POST con controllo campi obbligatori.
    \begin{verbatim}
app.use(express.json());

app.post('/tasks', (req, res) => {
    const newTask = req.body;
    if (!newTask.title) {
        return res.status(400).json({ error: "Required" });
    }
    tasks.push(newTask);
    res.status(201).json({ status: "created" });
});
    \end{verbatim}
\end{frame}

\section{Persistenza}
\begin{frame}{Strategie di Persistenza}
    Mantenere lo stato tra i riavvii del server.
    \begin{enumerate}
        \item \textbf{Lettura}: Caricamento del file JSON in memoria.
        \item \textbf{Modifica}: Manipolazione della struttura dati.
        \item \textbf{Scrittura}: Salvataggio su disco (operazione costosa).
    \end{enumerate}
    \vspace{0.2cm}
    \textbf{Problemi tecnici}:
    \begin{itemize}
        \item \textbf{Race Conditions}: Rischi di sovrascrittura con accessi concorrenti.
        \item \textbf{Atomicità}: Mancanza di transazioni sicure su file piatti.
    \end{itemize}
\end{frame}

\begin{frame}{Evoluzione: Dal File al Database}
    I database (RDBMS) risolvono i limiti della persistenza su file.
    \begin{itemize}
        \item \textbf{ACID}: Atomicità, Consistenza, Isolamento, Durabilità.
        \item \textbf{Concorrenza}: Gestione sicura di migliaia di richieste simultanee.
        \item \textbf{Esempi}: SQLite (lightweight), PostgreSQL (Enterprise).
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Persistenza JSON (Python)}
    \begin{verbatim}
def load_tasks():
    if not os.path.exists('tasks.json'): return []
    with open('tasks.json', 'r') as f:
        return json.load(f)

def save_tasks(tasks):
    with open('tasks.json', 'w') as f:
        json.dump(tasks, f, indent=4)
    \end{verbatim}
\end{frame}

\begin{frame}[fragile]{Persistenza JSON (Express)}
    Utilizzo di versioni sincrone (\texttt{readFileSync}) per semplicità didattica.
    \begin{verbatim}
const fs = require('fs');
const FILE_PATH = './tasks.json';

const loadTasks = () => {
    if (!fs.existsSync(FILE_PATH)) return [];
    return JSON.parse(fs.readFileSync(FILE_PATH));
};

const saveTasks = (tasks) => {
    fs.writeFileSync(FILE_PATH, JSON.stringify(tasks, null, 4));
};
    \end{verbatim}
\end{frame}

\section{Sicurezza}
\begin{frame}{Middleware Pipeline}
    I Middleware formano una catena di protezione sequenziale.
    \vspace{0.5cm}
    \centering
    \begin{tikzpicture}[
        node/.style={rectangle, draw, fill=red!10, minimum width=2cm, minimum height=0.8cm, align=center, font=\small},
        arrow/.style={-latex, thick, primary}
    ]
        \node (auth) [node] {Auth\\(401)};
        \node (val) [node, right=0.8cm of auth] {Validazione\\(400)};
        \node (log) [node, right=0.8cm of val] {Business\\Logic};
        \draw [arrow] (auth) -- (val);
        \draw [arrow] (val) -- (log);
    \end{tikzpicture}
    \begin{itemize}
        \item \textbf{Sanitizzazione}: Pulizia input obbligatoria prima della logica.
        \item \textbf{Isolamento}: Ogni middleware ha una sola responsabilità.
    \end{itemize}
\end{frame}

\begin{frame}{CORS: Cross-Origin Resource Sharing}
    Meccanismo di sicurezza del browser per le richieste cross-origin.
    \begin{itemize}
        \item \textbf{Problema}: Il browser blocca richieste tra domini diversi (es. \texttt{localhost:5500} $\rightarrow$ \texttt{localhost:3000}).
        \item \textbf{Soluzione}: Il server deve inviare header specifici per autorizzare l'origine.
    \end{itemize}
    \vspace{0.3cm}
    \texttt{Access-Control-Allow-Origin: *}
\end{frame}

\begin{frame}{Validazione e Sanitizzazione}
    La regola aurea: \textbf{Mai fidarsi dell'input del client}.
    \begin{itemize}
        \item \textbf{Validazione}: Verifica tipo e formato (es. email corretta).
        \item \textbf{Sanitizzazione}: Rimozione tag HTML/Script per prevenire XSS.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Basic Auth (Flask)}
    \begin{verbatim}
def check_auth(user, password):
    return user == 'admin' and password == 'secret'

@app.route('/secure')
def secure_endpoint():
    auth = request.authorization
    if not auth or not check_auth(auth.username, auth.password):
        return Response('Access denied', 401, 
           {'WWW-Authenticate': 'Basic realm="Login"'})
    return "Sensitive Data"
    \end{verbatim}
\end{frame}

\begin{frame}[fragile]{Basic Auth (Express)}

const basicAuth = require('express-basic-auth');

\begin{frame}[fragile]{Fetch API: Pattern Moderno}
    Gestione asincrona tramite \texttt{async/await} per leggibilità.
    \begin{verbatim}
async function getTasks() {
    try {
        const response = await fetch('/tasks');
        if (!response.ok) throw new Error('HTTP Error');
        const data = await response.json();
        renderTasks(data);
    } catch (err) {
        console.error('Request failed:', err);
    }
}
    \end{verbatim}
