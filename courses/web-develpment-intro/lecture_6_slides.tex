\section{WebSockets}
\begin{frame}{Protocollo WebSocket}
    \begin{itemize}
        \item \textbf{Bidirezionale}: Canale Full-Duplex.
        \item \textbf{Persistente}: Una volta stabilita (Handshake HTTP), non servono più header.
        \item \textbf{Librerie}: 
        \begin{itemize}
            \item \texttt{ws}: Standard-compliant, basica.
            \item \texttt{Socket.io}: Produzione-ready (auto-reconnect, rooms).
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Server WebSocket (ws)}
    \begin{lstlisting}[language=javascript]
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

wss.on('connection', (ws) => {
    ws.on('message', (msg) => {
        console.log('Ricevuto:', msg);
        ws.send('Messaggio ricevuto dal server');
    });
});
    \end{lstlisting}
\end{frame}

\section{WebRTC}
\begin{frame}{Architettura P2P}
    \begin{itemize}
        \item \textbf{P2P}: Dati audio/video diretti tra browser (bassa latenza).
        \item \textbf{Sfida NAT}: Dispositivi dietro firewall.
        \item \textbf{Signaling}: Necessario per scambiare SDP (metadati) via WebSocket.
        \item \textbf{Server Ice}:
        \begin{itemize}
            \item STUN: Rivelazione IP pubblico.
            \item TURN: Relè (se P2P fallisce).
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Setup RTCPeerConnection}
    \begin{lstlisting}[language=javascript]
const config = { 
    iceServers: [{ urls: 'stun:stun.l.google.com:19302' }] 
};
const pc = new RTCPeerConnection(config);

// Aggiunta traccia video/audio
localStream.getTracks().forEach(track => {
    pc.addTrack(track, localStream);
});

// Ricezione traccia remota
pc.ontrack = (event) => {
    remoteVideo.srcObject = event.streams[0];
};
    \end{lstlisting}
\end{frame}
